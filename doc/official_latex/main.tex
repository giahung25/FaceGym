\subsection{Lý do chọn đề tài}

Trong bối cảnh ngành fitness và gym đang phát triển mạnh mẽ tại Việt Nam, việc quản lý phòng gym theo cách truyền thống (sổ sách, Excel) gặp nhiều hạn chế:

\begin{itemize}[leftmargin=2cm]
	\item Khó theo dõi thông tin hội viên, gói tập, thiết bị khi số lượng lớn.
	\item Điểm danh thủ công tốn thời gian, dễ sai sót và không chuyên nghiệp.
	\item Không có công cụ thống kê, báo cáo để hỗ trợ ra quyết định kinh doanh.
	\item Hội viên không thể tự tra cứu thông tin gói tập, lịch sử của mình.
\end{itemize}

Đề tài này nhằm xây dựng một hệ thống quản lý phòng gym toàn diện, tích hợp công nghệ \textbf{nhận diện khuôn mặt (Face ID)} để tự động hóa điểm danh, nâng cao trải nghiệm người dùng và hiệu quả vận hành.

\subsection{Mục tiêu đề tài}

\begin{enumerate}[leftmargin=2cm]
	\item Xây dựng ứng dụng desktop quản lý phòng gym với đầy đủ chức năng CRUD cho hội viên, gói tập, thiết bị, huấn luyện viên.
	\item Tích hợp nhận diện khuôn mặt để điểm danh tự động thông qua camera.
	\item Cung cấp dashboard thống kê và báo cáo trực quan cho quản trị viên.
	\item Xây dựng app riêng cho hội viên và huấn luyện viên tự tra cứu thông tin.
	\item Áp dụng kiến trúc phân tầng (Layered Architecture) để dễ bảo trì và mở rộng.
\end{enumerate}

\section{Kết nối Python với CSDL}

\subsection{Context Manager \texttt{get\_db()}}

Hệ thống sử dụng Context Manager để quản lý kết nối database an toàn:

\begin{lstlisting}[style=pythonstyle, caption={Context Manager kết nối database}]
@contextmanager
def get_db():
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row     # Ket qua dang dict-like
    conn.execute("PRAGMA foreign_keys = ON")  # Bat Foreign Key
    try:
        yield conn        # Tra connection cho caller
        conn.commit()     # Khong loi -> commit
    except Exception:
        conn.rollback()   # Co loi -> rollback
        raise
    finally:
        conn.close()      # Luon dong connection
\end{lstlisting}

\textbf{Giải thích:}
\begin{itemize}[leftmargin=2cm]
	\item \texttt{@contextmanager}: Biến hàm thành context manager, dùng với \texttt{with}.
	\item \texttt{row\_factory = sqlite3.Row}: Truy cập cột bằng tên (\texttt{row["name"]}) thay vì index.
	\item \texttt{PRAGMA foreign\_keys = ON}: SQLite mặc định tắt FK, phải bật thủ công.
	\item Tự động \texttt{commit} khi thành công, \texttt{rollback} khi lỗi, \texttt{close} luôn luôn.
\end{itemize}

\subsection{Khởi tạo Database -- \texttt{init\_db()}}

def create_member(name, phone, email=None, gender=None,
                  date_of_birth=None, address=None,
                  emergency_contact=None):
    # Buoc 1: Validate du lieu dau vao

\begin{lstlisting}[style=pythonstyle, caption={Thêm hội viên -- Repository Layer}]
# app/repositories/member_repo.py
def create(member):
    with get_db() as conn:
        conn.execute("""
            INSERT INTO members
            (id, name, phone, email, gender, date_of_birth,
             address, emergency_contact, photo, pin,
             created_at, updated_at, is_active)
            VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?)
        """, (member.id, member.name, member.phone,
              member.email, member.gender,
              member.date_of_birth, member.address,
              member.emergency_contact, member.photo,
              member.pin,
              member.created_at.isoformat(),
              member.updated_at.isoformat(),
              int(member.is_active)))
    return member
\end{lstlisting}

\subsection{Xem (Read)}

\begin{lstlisting}[style=pythonstyle, caption={Lấy danh sách hội viên}]
# Repository: lay tat ca hoi vien dang active
def get_all(include_inactive=False):
    with get_db() as conn:
        if include_inactive:
            rows = conn.execute(
                "SELECT * FROM members ORDER BY created_at DESC"
            ).fetchall()
        else:
            rows = conn.execute(
                "SELECT * FROM members WHERE is_active = 1 "
                "ORDER BY created_at DESC"
            ).fetchall()
        return [_row_to_model(row) for row in rows]

# Chuyen row thanh object -- dung __new__ khong phai __init__
def _row_to_model(row):
    m = Member.__new__(Member)   # Khong goi __init__
    m.id = row["id"]             # Giu UUID tu DB
    m.name = row["name"]
    m.phone = row["phone"]
    m.created_at = datetime.fromisoformat(row["created_at"])
    m.is_active = bool(row["is_active"])  # 0/1 -> bool
    # ... cac field khac
    return m
\end{lstlisting}

\textbf{Tại sao dùng \texttt{\_\_new\_\_} thay vì \texttt{\_\_init\_\_}?} Vì \texttt{Member()} sẽ tạo UUID mới và ghi đè giá trị từ database.

\subsection{Xóa (Delete) -- Soft Delete}

\begin{lstlisting}[style=pythonstyle, caption={Xóa mềm hội viên (cascade)}]
# Repository: soft delete + cascade
def delete(member_id):
    with get_db() as conn:
        now = datetime.now().isoformat()
        # Soft delete member
        conn.execute(
            "UPDATE members SET is_active=0, updated_at=? "
            "WHERE id=?", (now, member_id))
        # Cascade: huy subscriptions
        conn.execute(
            "UPDATE subscriptions SET is_active=0, "
            "status='cancelled', updated_at=? "
            "WHERE member_id=?", (now, member_id))
        # Cascade: huy trainer assignments
        conn.execute(
            "UPDATE trainer_assignments SET is_active=0, "
            "status='inactive', updated_at=? "
            "WHERE member_id=?", (now, member_id))
        # Hard delete notifications
        conn.execute(
            "DELETE FROM notifications WHERE user_id=?",
            (member_id,))
\end{lstlisting}

\subsection{Nhận diện khuôn mặt (Face ID)}

Module Face ID sử dụng thư viện \texttt{face\_recognition} (dựa trên dlib) để mã hóa và nhận diện khuôn mặt:

\begin{lstlisting}[style=pythonstyle, caption={Mã hóa khuôn mặt}]
# app/face_id/face_encoder.py
import face_recognition
import numpy as np

class FaceEncoder:
    def encode_face(self, image):
        """Ma hoa khuon mat thanh vector 128 chieu"""
        # Detect vi tri khuon mat trong anh
        face_locations = face_recognition.face_locations(image)
        if not face_locations:
            return None

        # Ma hoa thanh vector so (128-dimensional embedding)
        encodings = face_recognition.face_encodings(
            image, face_locations
        )
        return encodings[0] if encodings else None
\end{lstlisting}

\begin{lstlisting}[style=pythonstyle, caption={Nhận diện khuôn mặt khi điểm danh}]
# app/face_id/face_recognizer.py
class FaceRecognizer:
    def __init__(self, tolerance=0.45):
        self.tolerance = tolerance  # Nguong tuong dong

    def recognize(self, face_encoding, known_encodings):
        """So sanh encoding voi database da biet"""
        for member_id, known_enc in known_encodings.items():
            # Tinh khoang cach Euclidean
            distance = face_recognition.face_distance(
                [known_enc], face_encoding
            )[0]

            if distance <= self.tolerance:
                confidence = 1.0 - distance
                return member_id, confidence

        return None, 0.0  # Khong nhan dien duoc
\end{lstlisting}

\textbf{Quy trình điểm danh:}
\begin{enumerate}[leftmargin=2cm]
	\item Camera chụp ảnh khuôn mặt hội viên.
	\item \texttt{FaceEncoder} mã hóa thành vector 128 chiều.
	\item \texttt{FaceRecognizer} so sánh với database encoding đã lưu.
	\item Nếu khoảng cách $\leq$ tolerance (0.45): nhận diện thành công $\rightarrow$ tự động check-in.
	\item Lưu vào bảng \texttt{attendance} với độ tin cậy (confidence).
\end{enumerate}

\subsection{Đăng ký khuôn mặt (Face Registration)}

\begin{lstlisting}[style=pythonstyle, caption={Đăng ký khuôn mặt cho hội viên}]
# app/face_id/face_register.py
class FaceRegistration:
    def register_member(self, member_id, images):
        """Dang ky khuon mat tu nhieu anh"""
        encodings = []
        for img in images:
            enc = self.encoder.encode_face(img)
            if enc is not None:
                encodings.append(enc)

        if not encodings:
            return False, "Khong phat hien khuon mat"

        # Lay encoding trung binh (robust hon)
        avg_encoding = np.mean(encodings, axis=0)

        # Luu vao file pickle
        self.save_encoding(member_id, avg_encoding)

        # Luu anh vao data/dataset/{member_id}/
        self.save_images(member_id, images)

        return True, "Dang ky thanh cong"
\end{lstlisting}

	\item \textbf{Ảnh gốc}: \texttt{data/dataset/\{member\_id\}/*.jpg} (5--10 ảnh mỗi người).
